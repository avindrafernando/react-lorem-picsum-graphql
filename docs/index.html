Let's make a query

Imports

import React, { Fragment }from "react";
import { useQuery } from "@apollo/react-hooks";
import gql from "graphql-tag";

MyComponent.js

Define Query

const GET_GITHUB_DATA = gql`
  query githubData($login: String!, $first: Int!) {
    organization(login: $login) {
      repositories(first: $first, privacy: PUBLIC) {
        totalCount
        nodes {
          id
          name
          url
          stargazers {
            totalCount
          }
        }
      }
    }
  }
`;

MyComponent.js

Query for Data

export default function MyComponent() {
  const { data, loading, error } = useQuery(GET_GITHUB_DATA, {
    variables: { login: "facebook", first: 50 }
  });

  if (loading) return <Loading />;
  if (error) return <Error message={`ERROR: Data failed to load`} />;

  return (
    <Fragment>
      ...
    </Fragment>
  );
}

cache.writeData({

index.js

Query local cache

const IS_AUTHENTICATED = gql`
  query isAuthenticatedWithGithub {
    isAuthenticated @client
  }
`;

function isAuthenticated() {
  const { data } = useQuery(IS_AUTHENTICATED);
  return data.isAuthenticated ? <MyComponent /> : <Error />;
}
